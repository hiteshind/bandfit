        function showSection(section) {

            document.querySelectorAll('[id^="section-"]').forEach(s => s.classList.add('hidden'));

            document.getElementById(`section-${section}`).classList.remove('hidden');

            

            // Update nav

            document.querySelectorAll('.nav-item').forEach(nav => nav.classList.remove('nav-active'));

            document.getElementById(`nav-${section}`).classList.add('nav-active');

            

            if (section === 'home') document.getElementById('page-title').textContent = "Welcome back, Champion";

            else if (section === 'exercises') document.getElementById('page-title').textContent = "All Exercises";

            else if (section === 'routines') document.getElementById('page-title').textContent = "Training Programs";

            else if (section === 'tracker') document.getElementById('page-title').textContent = "Progress Tracker";

        }

        function showExerciseDetail(exercise) {

            document.getElementById('modal-muscle').textContent = exercise.muscle;

            document.getElementById('modal-difficulty').textContent = exercise.difficulty;

            document.getElementById('modal-description').textContent = exercise.description;

            document.getElementById('modal-reps').textContent = exercise.reps;

        // Initialize filters

        function initFilters() {

            const muscles = ["All", "Legs", "Back", "Shoulders", "Arms", "Core"];

            const container = document.getElementById('muscle-filters');

            

            muscles.forEach(muscle => {

                const btn = document.createElement('button');

                btn.className = `px-6 py-3 rounded-2xl text-sm transition-all ${muscle === 'All' ? 'bg-green-500 text-black' : 'bg-zinc-800 hover:bg-zinc-700'}`;

                btn.textContent = muscle;

                btn.onclick = () => filterExercises(muscle);

                container.appendChild(btn);

            });

        }


        function filterExercises(muscle) {

            let filtered = exercises;

            

            if (muscle !== "All") {

                filtered = exercises.filter(ex => ex.muscle.toLowerCase().includes(muscle.toLowerCase()));

            }

            

            renderExercises('all-exercises', filtered);

        }

        }


        // Boot the app
